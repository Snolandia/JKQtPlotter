30/** \brief template class that wraps any datatype and combines it with a mutex, exposes the lock()/unlock()
31 * interface, so access to the contained data can be synchronized
32 * \ingroup jkqtptools_concurrency
33 *
34 *
35 */
36template <class T>
37class JKQTPSynchronized {
38public:
39 /** \brief Mutex used by this temmplate */
40 typedef std::mutex MutexType;
41 /** \brief type of a lock_guard for a JKQTPSynchronized<T> */
42 typedef std::lock_guard<JKQTPSynchronized<T> > Locker;
43 /** \brief contained data type T */
44 typedef T data_type;
45 /** \brief default constructor, the internal data is default-initialized */
46 inline JKQTPSynchronized(): m_data() {}
47 /** \brief initializing constructor, the internal data is initialized with \a d */
48 inline JKQTPSynchronized(const T& d): m_data(d) {}
49 /** \brief initializing constructor, the internal data is initialized with \a d */
50 inline JKQTPSynchronized(T&& d): m_data(std::forward<T>(d)) {}
51 /** \brief locks \a other and copies its contents to this new class */
52 inline JKQTPSynchronized(const JKQTPSynchronized<T>& other): m_data() {
53 Locker lock(other);
54 m_data=other.m_data;
55 }
56 /** \brief locks \a other and moves its contents to this new class. The mutex in \a other is NOT moved! */
57 inline JKQTPSynchronized(JKQTPSynchronized<T>&& other): m_data() {
58 Locker lock(other);
59 m_data=std::move(other.m_data);
60 }
61
62 /** \brief locks the internal mutex until unlock() is called,
63 *
64 * \note Use Locker instances to actually lock, using a RAII-idiom, as this is safer than doing this by hand!
65 */
66 inline void lock() {
67 m_mutex.lock();
68 }
69 /** \brief unlocks the internal mutex from a previous lock() call
70 *
71 * \note Use Locker instances to actually lock, using a RAII-idiom, as this is safer than doing this by hand!
72 */
73 inline void unlock() {
74 m_mutex.unlock();
75 }
76 /** \brief assign a value to the internal data storage, <b>not thread-safe.</b>
77 *
78 * \note You need to lock this object before calling this function.
79 */
80 inline void set(const T& d) {
81 m_data=d;
82 }
83 /** \brief assign a value to the internal data storage, <b>not thread-safe</b>.
84 *
85 * \note You need to lock this object before calling this function.
86 */
87 inline void set(T&& d) {
88 m_data=std::forward<T>(d);
89 }
90 /** \brief assign a value to the internal data storage, <b>thread-safe</b>.
91 */
92 inline void set_safe(const T& d) {
93 Locker lck(m_mutex);
94 m_data=d;
95 }
96 /** \brief assign a value to the internal data storage, \b thread-safe.
97 */
98 inline void set_safe(T&& d) {
99 Locker lck(m_mutex);
100 m_data=std::forward<T>(d);
101 }
102 /** \brief returns the internal data as a mutable reference, <b>not thread-safe</b>.
103 *
104 * \note You need to lock this object before calling this function.
105 */
106 inline T& get() {
107 return m_data;
108 }
109 /** \brief returns the internal data as a mutable reference, <b>not thread-safe</b>.
110 *
111 * \note You need to lock this object before calling this function.
112 */
113 inline const T& get() const {
114 return m_data;
115 }
116 /** \brief gives access to the internal data, <b>not thread-safe</b>.
117 *
118 * \note You need to lock this object before calling this function.
119 */
120 inline const T* operator->() const {
121 return &m_data;
122 }
123 /** \brief gives access to the internal data, <b>not thread-safe</b>.
124 *
125 * \note You need to lock this object before calling this function.
126 */
127 inline T* operator->() {
128 return &m_data;
129 }
130 /** \brief returns the value in the internal data storage, <b>thread-safe</b>.
131 */
132 inline T get_safe() const {
133 Locker lck(m_mutex);
134 return m_data;
135 }
136
137private:
138 T m_data;
139 mutable MutexType m_mutex;
140
141};
