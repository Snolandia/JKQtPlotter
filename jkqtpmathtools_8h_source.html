461/*! \brief evaluate a polynomial \f$ f(x)=\sum\limits_{i=0}^Pp_ix^i \f$ with \f$ p_i \f$ taken from the range \a firstP ... \a lastP
462 \ingroup jkqtptools_math_basic
463
464 \tparam PolyItP iterator for the polynomial coefficients
465 \param x where to evaluate
466 \param firstP points to the first polynomial coefficient \f$ p_1 \f$ (i.e. the offset with \f$ x^0 \f$ )
467 \param lastP points behind the last polynomial coefficient \f$ p_P \f$
468 \return value of polynomial \f$ f(x)=\sum\limits_{i=0}^Pp_ix^i \f$ at location \a x
469
470*/
471template <class PolyItP>
472inline double jkqtp_polyEval(double x, PolyItP firstP, PolyItP lastP) {
473 double v=0.0;
474 double xx=1.0;
475 for (auto itP=firstP; itP!=lastP; ++itP) {
476 v=v+(*itP)*xx;
477 xx=xx*x;
478 }
479 return v;
480}

509/*! \brief Generates a LaTeX string for the polynomial model with the coefficients \a firstP ... \a lastP
510 \ingroup jkqtptools_math_basic
511
512 \tparam PolyItP iterator for the polynomial coefficients
513 \param firstP points to the first polynomial coefficient \f$ p_1 \f$ (i.e. the offset with \f$ x^0 \f$ )
514 \param lastP points behind the last polynomial coefficient \f$ p_P \f$
515 */
516template <class PolyItP>
517QString jkqtp_polynomialModel2Latex(PolyItP firstP, PolyItP lastP) {
518 QString str="f(x)=";
519 size_t p=0;
520 for (auto itP=firstP; itP!=lastP; ++itP) {
521 if (p==0) str+=jkqtp_floattolatexqstr(*itP, 3);
522 else {
523 if (*itP>=0) str+="+";
524 str+=QString("%2{\\cdot}x^{%1}").arg(p).arg(jkqtp_floattolatexqstr(*itP, 3));
525 }
526 p++;
527 }
528 return str;
529}
