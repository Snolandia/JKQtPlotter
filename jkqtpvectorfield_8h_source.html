41/*! \brief This graph plots a vector field, i.e. a set of vectors (dx,dy) or (angle,length) at positions (x,y).
42 This class immplements the most basic form of vector plot, i.e. the vector are drawn with a length
43 corresponding to their magnitude.
44 \ingroup jkqtplotter_vectorfieldgraphs
45
46 \note This type of plot is sometimes also refered to as quiver plot (e.g. in Matlab or matplotlib)
47
48 \image html JKQTPVectorFieldGraph.png
49
50 To achieve this, use code like this:
51 \code
52 // 1. setup a plotter window and get a pointer to the internal datastore (for convenience)
53 JKQTPlotter plot;
54 JKQTPDatastore* ds=plot.getDatastore();
55
56 // 2. make up some arbitrary data to be used for plotting
57 // this generates a 2D grid of x/y-coordinates and then calculates dx=cos(y) and dy=sin(x)
58 const auto columnXY=ds->addLinearGridColumns(NX, 0, 6, NY, -3, 3,"x","y");
59 const auto columnDX=ds->addCalculatedColumnFromColumn(columnXY.first, columnXY.second, [](double x,double y) { return sin(y); });
60 const auto columnDY=ds->addCalculatedColumnFromColumn(columnXY.first, columnXY.second, [](double x,double y) { return cos(x); });
61
62 // 3. create JKQTPVectorFieldGraph to display the data:
63 JKQTPVectorFieldGraph* graph1=new JKQTPVectorFieldGraph(&plot);
64 graph1->setXYColumns(columnXY);
65 graph1->setDxColumn(columnDX);
66 graph1->setDyColumn(columnDY);
67 graph1->setTitle(QObject::tr("$\\vec{f}(x,y)=\\bigl[\\sin(y), \\cos(x)\\bigr]^\\mathrm{T}$"));
68
69 // 4. add the graphs to the plot, so it is actually displayed
70 plot.addGraph(graph1);
71 \endcode
72
73 \see \ref JKQTPlotterVectorFieldExample , JKQTPGraphDecoratedLineStyleMixin , JKQTPXYAndVectorGraph
74
75 */
76class JKQTPLOTTER_LIB_EXPORT JKQTPVectorFieldGraph: public JKQTPXYAndVectorGraph, public JKQTPGraphDecoratedLineStyleMixin {
77 Q_OBJECT
78 public:
79 enum VectorAnchorPoint {
80 AnchorBottom, //!< \brief this is the default: the vector starts at (x,y) \image html JKQTPVectorFieldGraphAnchorBottom.png
81 AnchorMid, //!< \brief the vector's mid is at (x,y) \image html JKQTPVectorFieldGraphAnchorMid.png
82 AnchorTip //!< \brief the vector ends at (x,y) \image html JKQTPVectorFieldGraphAnchorTip.png
83 };
84 Q_ENUM(VectorAnchorPoint)

121 Q_PROPERTY(double lengthScaleFactor READ getLengthScaleFactor WRITE setLengthScaleFactor )

123 protected:
124 private:
125 /** \brief enables or disables the autoscaling of vector lengths
126 *
127 * If disabled (\c false ) the vector is drawn from \c (x,y) to \c (x+dx,y+dy)*m_lengthScaleFactor ,
128 * otherweise to \c (x+dx,y+dy)*autoscale*m_autoscaleLengthFactor .
129 * The autoscaled length is calculated by a siple algorithm that uses the average vector length in the data:
130 * \c autoscale=plotwidth/VectorPerWidth/avgVectorLength .
131 *
132 * \see m_autoscaleFactor, m_autoscaleLengthFactor, setAutoscaleLength(), getAutoscaleLength()
133 */
134 bool m_autoscaleLength;
135 /** \brief a scaling factor that can be used to modify the result of the autoscaling algorithm (m_autoscaleLength \c ==true)
136 *
137 * The vector length is further scaled by this value.
138 * \see m_autoscaleLength, setAutoscaleFactor(), getAutoscaleFactor()
139 */
140 double m_autoscaleLengthFactor;
141 /** \brief if m_autoscaleLength \c ==false, this is the scale-factor used to calculate the vector length
142 *
143 * \see setLengthScaleFactor(), getLengthScaleFactor(), m_autoscaleLength
144 */
145 double m_lengthScaleFactor;
146 /** \brief defines where the vector is anchored
147 *
148 * \see VectorAnchorPoint
149 */
150 VectorAnchorPoint m_anchorPoint;
