33/** \brief this class can be used to implement a general cache for values
34 * \ingroup jkqtptools_concurrency
35 *
36 * It is typically used to generate a static (ThreadSafe=true) of thread_local (ThreadSafe=false) cache inside a function.
37 *
38 * The class is parametrized by a key (TKeay) and value (TData) data type and receives (in the constructor) a functor that
39 * calculates the data for a key. An additional template parameter ThreadSafe indicates (true|false) whether the class
40 * is implemented in a thread-safe way (for static instances) or not (then it should be used as thread_local instances in a
41 * multi-threaded environment or in a single-thread environment).
42 *
43 * The class member function get(key) returns a value for a given key, which is either taken from the internal cache,
44 * or generated using the functor provided to the constructor. In the latter case the generated value is stored in the internal cache.
45 *
46 * Internally the cache maps TKey to TData, but the signature of the get()-function and the generator functor actually uses TKeyInSignature,
47 * which may differ from TKey. The only limitation is that TKeyInSignature can be converted/assigned to a TKey
48 */
49template <class TData, class TKey, bool ThreadSafe=true, class TKeyInSignature=TKey>
50struct JKQTPDataCache {
51 template <typename FF>
52 inline JKQTPDataCache(FF generateData):
53 m_generateData(std::forward<FF>(generateData))
54 {
55
56 }
57
58 template<class... Args>
59 inline TData get_inline(Args... args) {
60 return get(TKeyInSignature(args...));
61 }
62
63 template <bool TSS=ThreadSafe>
64 inline TData get(const typename std::enable_if<TSS, TKeyInSignature>::type& key) {
65 const TKey cacheKey=key;
66
67 QReadLocker lockR(&m_mutex);
68 if (m_cache.contains(cacheKey)) return m_cache[cacheKey];
69 lockR.unlock();
70
71 QWriteLocker lockW(&m_mutex);
72 if (m_cache.contains(cacheKey)) return m_cache[cacheKey];
73
74 const auto newData=m_generateData(key);
75 m_cache[cacheKey]=newData;
76 return newData;
77 }
78
79 template <bool TSS=ThreadSafe>
80 inline TData get(const typename std::enable_if<!TSS, TKeyInSignature>::type& key) {
81 const TKey cacheKey=key;
82
83 if (m_cache.contains(cacheKey)) return m_cache[cacheKey];
84 const auto newData=m_generateData(key);
85 m_cache[cacheKey]=newData;
86 return newData;
87 }
88private:
89 QHash<TKey, TData> m_cache;
90 mutable typename std::enable_if<ThreadSafe, QReadWriteLock>::type m_mutex;
91 const std::function<TData(TKeyInSignature)> m_generateData;
92 Q_DISABLE_COPY(JKQTPDataCache)
93};
